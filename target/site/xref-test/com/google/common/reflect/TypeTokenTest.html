282   private static void assertSubtypeFirst(TypeToken<?>.TypeSet types) {
283     assertSubtypeTokenBeforeSupertypeToken(types);
284     assertSubtypeTokenBeforeSupertypeToken(types.interfaces());
285     assertSubtypeTokenBeforeSupertypeToken(types.classes());
286     assertSubtypeBeforeSupertype(types.rawTypes());
287     assertSubtypeBeforeSupertype(types.interfaces().rawTypes());
288     assertSubtypeBeforeSupertype(types.classes().rawTypes());
289   }

291   private static void assertSubtypeTokenBeforeSupertypeToken(
292       Iterable<? extends TypeToken<?>> types) {
293     int i = 0;
294     for (TypeToken<?> left : types) {
295       int j = 0;
296       for (TypeToken<?> right : types) {
297         if (left.isAssignableFrom(right)) {
298           assertTrue(left + " should be after " + right, i >= j);
299         }
300         j++;
301       }
302       i++;
303     }
304   }

306   private static void assertSubtypeBeforeSupertype(Iterable<? extends Class<?>> types) {
307     int i = 0;
308     for (Class<?> left : types) {
309       int j = 0;
310       for (Class<?> right : types) {
311         if (left.isAssignableFrom(right)) {
312           assertTrue(left + " should be after " + right, i >= j);
313         }
314         j++;
315       }
316       i++;
317     }
318   }
