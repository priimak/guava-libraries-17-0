144 /**
145  * Supplies an arbitrary "default" instance for a wide range of types, often useful in testing
146  * utilities.
147  * 
148  * <p>Covers arrays, enums and common types defined in {@code java.lang}, {@code java.lang.reflect},
149  * {@code java.io}, {@code java.nio}, {@code java.math}, {@code java.util}, {@code
150  * java.util.concurrent}, {@code java.util.regex}, {@code com.google.common.base}, {@code
151  * com.google.common.collect} and {@code com.google.common.primitives}. In addition, if the type
152  * exposes at least one public static final constant of the same type, one of the constants will be
153  * used; or if the class exposes a public parameter-less constructor then it will be "new"d and
154  * returned.
155  * 
156  * <p>All default instances returned by {@link #get} are generics-safe. Clients won't get type
157  * errors for using {@code get(Comparator.class)} as a {@code Comparator<Foo>}, for example.
158  * Immutable empty instances are returned for collection types; {@code ""} for string;
159  * {@code 0} for number types; reasonable default instance for other stateless types. For mutable
160  * types, a fresh instance is created each time {@code get()} is called.
161  *
162  * @author Kevin Bourrillion
163  * @author Ben Yu
164  * @since 12.0
165  */
166 @Beta
167 public final class ArbitraryInstances {

266   /**
267    * type -> implementation. Inherently mutable interfaces and abstract classes are mapped to their
268    * default implementations and are "new"d upon get().
269    */
270   private static final ConcurrentMap<Class<?>, Class<?>> implementations = Maps.newConcurrentMap();

272   private static <T> void setImplementation(Class<T> type, Class<? extends T> implementation) {
273     checkArgument(type != implementation, "Don't register %s to itself!", type);
274     checkArgument(!DEFAULTS.containsKey(type),
275         "A default value was already registered for %s", type);
276     checkArgument(implementations.put(type, implementation) == null,
277         "Implementation for %s was already registered", type);
278   }

310   /**
311    * Returns an arbitrary instance for {@code type}, or {@code null} if no arbitrary instance can
312    * be determined.
313    */
314   @Nullable public static <T> T get(Class<T> type) {
315     T defaultValue = DEFAULTS.getInstance(type);
316     if (defaultValue != null) {
317       return defaultValue;
318     }
319     Class<? extends T> implementation = getImplementation(type);
320     if (implementation != null) {
321       return get(implementation);
322     }
323     if (type.isEnum()) {
324       T[] enumConstants = type.getEnumConstants();
325       return (enumConstants.length == 0)
326           ? null
327           : enumConstants[0];
328     }
329     if (type.isArray()) {
330       return createEmptyArray(type);
331     }
332     T jvmDefault = Defaults.defaultValue(Primitives.unwrap(type));
333     if (jvmDefault != null) {
334       return jvmDefault;
335     }
336     if (Modifier.isAbstract(type.getModifiers()) || !Modifier.isPublic(type.getModifiers())) {
337       return arbitraryConstantInstanceOrNull(type);
338     }
339     final Constructor<T> constructor;
340     try {
341       constructor = type.getConstructor();
342     } catch (NoSuchMethodException e) {
343       return arbitraryConstantInstanceOrNull(type);
344     }
345     constructor.setAccessible(true); // accessibility check is too slow
346     try {
347       return constructor.newInstance();
348     } catch (InstantiationException impossible) {
349       throw new AssertionError(impossible);
350     } catch (IllegalAccessException impossible) {
351       throw new AssertionError(impossible);
352     } catch (InvocationTargetException e) {
353       logger.log(Level.WARNING, "Exception while invoking default constructor.", e.getCause());
354       return arbitraryConstantInstanceOrNull(type);
355     }
356   }

358   @Nullable private static <T> T arbitraryConstantInstanceOrNull(Class<T> type) {
359     Field[] fields = type.getDeclaredFields();
360     Arrays.sort(fields, BY_FIELD_NAME);
361     for (Field field : fields) {
362       if (Modifier.isPublic(field.getModifiers())
363           && Modifier.isStatic(field.getModifiers())
364           && Modifier.isFinal(field.getModifiers())) {
365         if (field.getGenericType() == field.getType()
366             && type.isAssignableFrom(field.getType())) {
367           field.setAccessible(true);
368           try {
369             T constant = type.cast(field.get(null));
370             if (constant != null) {
371               return constant;
372             }
373           } catch (IllegalAccessException impossible) {
374             throw new AssertionError(impossible);
375           }
376         }
377       }
378     }
379     return null;
380   }

382   private static <T> T createEmptyArray(Class<T> arrayType) {
383     return arrayType.cast(Array.newInstance(arrayType.getComponentType(), 0));
384   }

435   // Compare by toString() to satisfy 2 properties:

438   private static final class ByToString implements Comparable<Object>, Serializable {
