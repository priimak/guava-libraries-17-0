21  import com.google.common.base.Throwables;
22  import com.google.common.collect.ImmutableList;

24  import java.util.Collection;
25  import java.util.List;
26  import java.util.concurrent.Callable;
27  import java.util.concurrent.ExecutionException;
28  import java.util.concurrent.ExecutorService;
29  import java.util.concurrent.Executors;
30  import java.util.concurrent.Future;
31  import java.util.concurrent.TimeUnit;
32  import java.util.concurrent.TimeoutException;

34  /**
35   * An abstract {@code ExecutorService} that allows subclasses to
36   * {@linkplain #wrapTask(Callable) wrap} tasks before they are submitted
37   * to the underlying executor.
38   *
39   * <p>Note that task wrapping may occur even if the task is never executed.
40   *
41   * <p>For delegation without task-wrapping, see
42   * {@link ForwardingExecutorService}.
43   *
44   * @author Chris Nokleberg
45   */
46  abstract class WrappingExecutorService implements ExecutorService {
47    private final ExecutorService delegate;
48  
49    protected WrappingExecutorService(ExecutorService delegate) {
50      this.delegate = checkNotNull(delegate);
51    }
52  
53    /**
54     * Wraps a {@code Callable} for submission to the underlying executor. This
55     * method is also applied to any {@code Runnable} passed to the default
56     * implementation of {@link #wrapTest(Runnable)}.
57     */
58    protected abstract <T> Callable<T> wrapTask(Callable<T> callable);
59  
60    /**
61     * Wraps a {@code Runnable} for submission to the underlying executor. The
62     * default implementation delegates to {@link #wrapTask(Callable)}.
63     */
64    protected Runnable wrapTask(Runnable command) {
65      final Callable<Object> wrapped = wrapTask(
66          Executors.callable(command, null));
67      return new Runnable() {
68        @Override public void run() {
69          try {
70            wrapped.call();
71          } catch (Exception e) {
72            Throwables.propagate(e);
73          }
74        }
75      };
76    }
77  
78    /**
79     * Wraps a collection of tasks.
80     *
81     * @throws NullPointerException if any element of {@code tasks} is null
82     */
83    private final <T> ImmutableList<Callable<T>> wrapTasks(
84        Collection<? extends Callable<T>> tasks) {
85      ImmutableList.Builder<Callable<T>> builder = ImmutableList.builder();
86      for (Callable<T> task : tasks) {
87        builder.add(wrapTask(task));
88      }
89      return builder.build();
90    }
91  
92    // These methods wrap before delegating.
93    @Override
94    public final void execute(Runnable command) {
95      delegate.execute(wrapTask(command));
96    }
97  
98    @Override
99    public final <T> Future<T> submit(Callable<T> task) {
100     return delegate.submit(wrapTask(checkNotNull(task)));
101   }
102 
103   @Override
104   public final Future<?> submit(Runnable task) {
105     return delegate.submit(wrapTask(task));
106   }
107 
108   @Override
109   public final <T> Future<T> submit(Runnable task, T result) {
110     return delegate.submit(wrapTask(task), result);
111   }
112 
113   @Override
114   public final <T> List<Future<T>> invokeAll(
115       Collection<? extends Callable<T>> tasks) throws InterruptedException {
116     return delegate.invokeAll(wrapTasks(tasks));
117   }
118 
119   @Override
120   public final <T> List<Future<T>> invokeAll(
121       Collection<? extends Callable<T>> tasks, long timeout, TimeUnit unit)
122       throws InterruptedException {
123     return delegate.invokeAll(wrapTasks(tasks), timeout, unit);
124   }
125 
126   @Override
127   public final <T> T invokeAny(Collection<? extends Callable<T>> tasks)
128       throws InterruptedException, ExecutionException {
129     return delegate.invokeAny(wrapTasks(tasks));
130   }
131 
132   @Override
133   public final <T> T invokeAny(
134       Collection<? extends Callable<T>> tasks, long timeout, TimeUnit unit)
135       throws InterruptedException, ExecutionException, TimeoutException {
136     return delegate.invokeAny(wrapTasks(tasks), timeout, unit);
137   }
138 
139   // The remaining methods just delegate.
140 
141   @Override
142   public final void shutdown() {
143     delegate.shutdown();
144   }
145 
146   @Override
147   public final List<Runnable> shutdownNow() {
148     return delegate.shutdownNow();
149   }
150 
151   @Override
152   public final boolean isShutdown() {
153     return delegate.isShutdown();
154   }
155 
156   @Override
157   public final boolean isTerminated() {
158     return delegate.isTerminated();
159   }
160 
161   @Override
162   public final boolean awaitTermination(long timeout, TimeUnit unit)
163       throws InterruptedException {
164     return delegate.awaitTermination(timeout, unit);
165   }
166 }
